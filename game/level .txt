    <script>
        function goBack() {
            // 根据当前关卡返回到对应的学科关卡页面
            const urlParams = new URLSearchParams(window.location.search);
            const subject = urlParams.get('subject') || 'chemistry';
            window.location.href = 'subject-levels.html?subject=' + subject;
        }

        // 页面加载时根据URL参数设置学科和关卡信息
        window.onload = function() {
            const urlParams = new URLSearchParams(window.location.search);
            const subject = urlParams.get('subject');
            const level = urlParams.get('level') || '1';
            
            if (subject) {
                const subjectNames = {
                    'biology': '生物',
                    'physics': '物理',
                    'geography': '地理',
                    'chemistry': '化学',
                    'history': '历史',
                    'astronomy': '天文'
                };
                
                const subjectName = subjectNames[subject] || '学科';
                document.querySelector('.level-info').textContent = subjectName + '-关卡' + level;
            }
        }

        // 提交答案按钮事件
        document.querySelector('.submit-button').addEventListener('click', function() {
            const answer = document.querySelector('.answer-input').value;
            if (answer.trim() === '') {
                alert('请输入答案');
            } else {
                alert('提交答案：' + answer + '\n（此处后期会与后端交互）');
            }
        });
    </script>
